-------------Inserir Atendimentos
curl -i -X POST http://localhost:3001/atendimentos \
     -H 'Content-Type: application/json' \
     -d $'{
            "cliente": "Maria Joana", 
            "pet": "Stopinha", 
	    "servico": "Banho",
	    "data": "30/04/2022",
            "Status": "Agendado",
            "observacoes": "De acordo ao manual do curl ( disponível online [...]"
          }' 
-----------------------------------------------------------------

-------------Editar Atendimentos
curl -i -X PATCH http://localhost:3001/atendimentos/4 \
     -H 'Content-Type: application/json' \
     -d $'{	    
            "cliente": "Maria Joanaquina", 
            "pet": "Stopinha", 
	    "servico": "Banho",
	    "data": "30/04/2022",
            "Status": "Agendado",
            "observacoes": "De acordo ao manual do curl ( disponível online [...]"
          }'           
-----------------------------------------------------------------

-------------Editar Atendimentos
curl -i -X DELETE http://localhost:3001/atendimentos/4 \
     -H 'Content-Type: application/json' \
     
-----------------------------------------------------------------

-------------Consulta Atendimentos
curl -i -X GET http://localhost:3001/atendimentos/ \
     -H 'Content-Type: application/json' \
     
-----------------------------------------------------------------

-------------Consulta Atendimento especifico
curl -i -X GET http://localhost:3001/atendimentos/1 \
     -H 'Content-Type: application/json' \
